SFTt_CAN_USE_X_SLOT = {

	OR = {

		AND = { has_technology = tech_energy_lance_1 NOT = { no_resource_for_component = { RESOURCE = rare_crystals } } }
		AND = { has_technology = tech_energy_lance_2 NOT = { no_resource_for_component = { RESOURCE = rare_crystals } } }
		AND = { has_technology = tech_arc_emitter_1 NOT = { no_resource_for_component = { RESOURCE = rare_crystals } } }
		AND = { has_technology = tech_arc_emitter_2 NOT = { no_resource_for_component = { RESOURCE = rare_crystals } } }
		AND = { has_technology = tech_mass_accelerator_1 NOT = { no_resource_for_component = { RESOURCE = volatile_motes } } }
		AND = { has_technology = tech_mass_accelerator_2 NOT = { no_resource_for_component = { RESOURCE = volatile_motes } } }

		##### 고대 가속기는 고려하지 않음. 자원 문제에 AI 가 대응하지 못할 가능성이 너무 높음.
		##### 바이오쉽도 기술이름과 자원종류는 똑같으니까 그대로 쓰면 됨

	}

}

SFTt_END_GAME_ENKI = {

	AND = {

		NOT = { has_country_flag = SFTf_DESIGN_UPDATED }
		SFTt_CAN_USE_X_SLOT = yes

		OR = {
			is_preferred_weapons = weapon_type_energy
			is_preferred_weapons = weapon_type_kinetic
		}

	}

	##### EN/KI 선호, X 슬롯 사용 가능한데 업데이트 쿨다운마다 돌리고 있다면 최후반으로 평가해서 가중치 추가

}
