col_city = {
	icon = GFX_colony_type_urban

	potential = {
		owner_or_space_owner? = {
			is_regular_empire = yes
			is_nomadic = no
		}
		NOR = {
			uses_district_set = city_world

		}

		is_capital = no
		NOT = { is_planet_class = pc_nanotech }

		inline_script = "stellarai/invalid_urban_colony_type_potential"

	inline_script = {
		script = colony_types/colony_type_planet_modifier
		MODIFIER = trader_jobs_bonus_workforce_mult
	}

	ai_building_set_affinity = {
		trade = 2.0
	}

	weight_modifier = {
		base = 10
		modifier = {
			add = @stellarai_easy_to_swap
			has_designation = col_city
		}
		modifier = {
			num_zones = {
				type = zone_trade
				value > 0
			}
			add = @quite_low_zones_weight # intended, this is not a particularly great colony designation
		}
		inline_script = {
			script = colony_types/shared_weights_urban
		}
		inline_script = "stellarai/urban_penalty_commitment_flags" 
		inline_script = "stellarai/urban_not_active_penalty"
	}
}
