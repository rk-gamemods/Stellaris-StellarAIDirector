modifier = {
    # Hard guard: do not keep adding upkeep-heavy buildings when the empire is already in a strategic resource shortage.
    factor = 0.001
    owner = {
        OR = {
            has_deficit = $STRATEGIC_RESOURCE$
            has_monthly_income = {
                resource = $STRATEGIC_RESOURCE$
                value < 0
            }
        }
    }
}

modifier = {
    # Strong pre-deficit guard: low stockpile and weak income should slow further strategic upkeep growth before a crisis begins.
    factor = 0.12
    owner = {
        AND = {
            resource_stockpile_compare = {
                resource = $STRATEGIC_RESOURCE$
                value < 60
            }
            NOT = {
                has_monthly_income = {
                    resource = $STRATEGIC_RESOURCE$
                    value > $MIN_INCOME$
                }
            }
        }
    }
}

modifier = {
    # Existing local guard, kept but strengthened slightly by the two checks above.
    factor = 0.01
    owner = {
        OR = {
            has_monthly_income = {
                resource = $STRATEGIC_RESOURCE$
                value < $MIN_INCOME$
            }
            resource_stockpile_compare = {
                resource = $STRATEGIC_RESOURCE$
                value < $MIN_STOCKPILE$
            }
        }
    }
}

modifier = {
    factor = $EARLY_FACTOR$
    years_passed < $EARLY_YEARS$
    owner = {
        resource_stockpile_compare = {
            resource = $STRATEGIC_RESOURCE$
            value < $EARLY_STOCKPILE$
        }
    }
}

modifier = {
    # Only relax the guard when both income and stockpile are clearly safe.
    factor = $SURPLUS_FACTOR$
    owner = {
        AND = {
            has_monthly_income = {
                resource = $STRATEGIC_RESOURCE$
                value > $SURPLUS_INCOME$
            }
            resource_stockpile_compare = {
                resource = $STRATEGIC_RESOURCE$
                value >= $SURPLUS_STOCKPILE$
            }
        }
    }
}
