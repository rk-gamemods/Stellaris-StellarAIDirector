stellarai_military_emergency = {
	OR = {
		AND = {
			highest_threat > 30
			used_naval_capacity_percent < 0.90
		}
		AND = {
			is_at_war = yes
			highest_threat > 20
			used_naval_capacity_percent < 1.00
		}
		AND = {
			OR = {
				has_valid_civic = civic_machine_terminator
				has_valid_civic = civic_fanatic_purifiers
				has_valid_civic = civic_hive_devouring_swarm
			}
			highest_threat > 24
			used_naval_capacity_percent < 1.05
		}
		highest_threat > 45
	}
}

# Shared settled-empire economic states. These are the single source of truth
# for economic plans, research construction, designations, and fleet spending.
stellarai_economic_crisis = {
	is_nomadic = no
	OR = {
		has_deficit = energy
		has_deficit = minerals
		resource_stockpile_compare = { resource = energy value < 500 }
		resource_stockpile_compare = { resource = minerals value < 500 }
		AND = {
			country_uses_food = yes
			OR = {
				has_deficit = food
				resource_stockpile_compare = { resource = food value < 500 }
			}
		}
		AND = {
			country_uses_consumer_goods = yes
			OR = {
				has_deficit = consumer_goods
				resource_stockpile_compare = { resource = consumer_goods value < 300 }
			}
		}
	}
}

stellarai_consumer_goods_pressure = {
	is_nomadic = no
	country_uses_consumer_goods = yes
	OR = {
		has_deficit = consumer_goods
		NOT = { has_monthly_income = { resource = consumer_goods value > 8 } }
		resource_stockpile_compare = { resource = consumer_goods value < 1200 }
	}
}

stellarai_foundation_ready = {
	is_nomadic = no
	NOT = { stellarai_economic_crisis = yes }
	NOT = { has_deficit = energy }
	NOT = { has_deficit = minerals }
	has_monthly_income = { resource = energy value > 15 }
	has_monthly_income = { resource = minerals value > 12 }
	resource_stockpile_compare = { resource = energy value > 1500 }
	resource_stockpile_compare = { resource = minerals value > 1000 }
	OR = {
		country_uses_food = no
		AND = {
			NOT = { has_deficit = food }
			has_monthly_income = { resource = food value > 5 }
			resource_stockpile_compare = { resource = food value > 750 }
		}
	}
	OR = {
		country_uses_consumer_goods = no
		AND = {
			NOT = { has_deficit = consumer_goods }
			has_monthly_income = { resource = consumer_goods value > 6 }
			resource_stockpile_compare = { resource = consumer_goods value > 750 }
		}
	}
}

stellarai_research_ready = {
	years_passed > 44
	is_nomadic = no
	NOT = { stellarai_economic_crisis = yes }
	NOT = { has_deficit = energy }
	NOT = { has_deficit = minerals }
	has_monthly_income = { resource = energy value > 25 }
	has_monthly_income = { resource = minerals value > 20 }
	resource_stockpile_compare = { resource = energy value > 2500 }
	resource_stockpile_compare = { resource = minerals value > 1500 }
	OR = {
		country_uses_food = no
		AND = {
			NOT = { has_deficit = food }
			has_monthly_income = { resource = food value > 10 }
			resource_stockpile_compare = { resource = food value > 1000 }
		}
	}
	OR = {
		country_uses_consumer_goods = no
		AND = {
			NOT = { has_deficit = consumer_goods }
			has_monthly_income = { resource = consumer_goods value > 10 }
			resource_stockpile_compare = { resource = consumer_goods value > 1200 }
		}
	}
}

stellarai_should_rebuild_fleet = {
	is_nomadic = no
	country_uses_bio_ships = no
	OR = {
		AND = {
			is_at_war = yes
			used_naval_capacity_percent < 0.85
		}
		AND = {
			stellarai_military_emergency = yes
			used_naval_capacity_percent < 0.85
		}
		AND = {
			is_at_war = no
			years_passed < 45
			NOT = { stellarai_economic_crisis = yes }
			used_naval_capacity_percent < 0.60
		}
		AND = {
			is_at_war = no
			years_passed > 44
			stellarai_research_ready = yes
			used_naval_capacity_percent < 0.85
		}
		AND = {
			is_at_war = no
			years_passed > 44
			stellarai_foundation_ready = yes
			used_naval_capacity_percent < 0.70
		}
		AND = {
			recently_lost_war = yes
			stellarai_foundation_ready = yes
			used_naval_capacity_percent < 0.75
		}
	}
}
