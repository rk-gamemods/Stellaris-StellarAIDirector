alloys_expenditure_ships = {
	resource = alloys
	type = expenditure
	category = ships

	potential = {
		always = yes
	}

	weight = {
		weight = 1.0

		modifier = {
			factor = 4.0
			stellarai_should_rebuild_fleet = yes
		}

		# Protect early peacetime colonization from the stronger navy/alloy budget.
		# If the AI has colony plans and no real threat, expansion should win over fleet buildup.
		modifier = {
			factor = 0.12
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
				highest_threat < 20
				is_machine_empire = no
				NOT = { stellarai_military_emergency = yes }
			}
		}

		modifier = {
			factor = 0.30
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
				highest_threat < 20
				is_machine_empire = yes
				NOT = { has_valid_civic = civic_machine_terminator }
				NOT = { stellarai_military_emergency = yes }
			}
		}

		modifier = {
			factor = 0.60
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
				highest_threat < 20
				has_valid_civic = civic_machine_terminator
				NOT = { stellarai_military_emergency = yes }
			}
		}

		# spend less alloys on ships if over capacity
		modifier = {
			factor = 0.45
			is_at_war = no
			stellarai_machine_energy_emergency = yes
		}

		modifier = {
			factor = 0.85
			is_at_war = no
			stellarai_machine_minerals_emergency = yes
		}

		modifier = {
			factor = 1.15
			stellarai_machine_basic_econ_stable = yes
			used_naval_capacity_percent < 1.0
		}

		modifier = {
			factor = 0.60
			is_at_war = no
			stellarai_economic_crisis = yes
		}

		modifier = {
			factor = 0.65
			AND = {
				used_naval_capacity_percent >= 1.0
				is_at_war = no
			}
		}

		modifier = {
			factor = 0.82
			AND = {
				used_naval_capacity_percent >= 1.15
				is_at_war = yes
			}
		}

		# Biological Ships use less alloys
		modifier = {
			factor = 0.33
			country_uses_bio_ships = yes
		}

		modifier = {
			factor = 5
			used_naval_capacity_percent < 1.5
			OR = {
				any_neighbor_country = {
					has_ascension_perk = ap_become_the_crisis
				}
				any_country = {
					is_crisis_faction = yes
				}
			}
		}
	
	}

	desired_min = {
		base = 0

		modifier = {
			add = @ai_ship_budget_desired_min_per_year
			country_uses_bio_ships = no
		}

		modifier = {
			factor = trigger:years_passed
		}
	}
}

			factor = 2.5

		# Convert a stable research lead into current military power.
		modifier = {
			factor = 1.50

		}

		# Biological Ships use less alloys
		modifier = {
			factor = 0.33
			country_uses_bio_ships = yes
		}
	}
}

		# Reduce spending on claiming new systems if we have unclaimed planets
		modifier = {
			is_machine_empire = no
			has_country_resource = { type = alloys amount < 250 }

		}

		modifier = {
			factor = 1.20

		}

			factor = 0.45

			factor = 1.35

				can_be_upgraded = yes
			}
		}

		modifier = {
			factor = 1.25
			highest_threat > 20
			OR = {
				any_owned_starbase = {
					can_be_upgraded = yes

					can_be_upgraded = yes

			factor = 0.5

			factor = 2
			is_nomadic = yes

}

			}
		}
		# Bioship Starhold costs 500 alloys - base 500 is sufficient, no add needed

				has_starbase_size = starbase_starhold
				can_be_upgraded = yes
			}
		}
		modifier = {
