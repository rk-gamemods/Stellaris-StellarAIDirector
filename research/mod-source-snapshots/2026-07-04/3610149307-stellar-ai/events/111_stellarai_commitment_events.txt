      limit = {
        OR = {
          is_regular_empire = yes

          is_galvanic_empire = yes
        }
      }
      country_event = { id = stellarai_commitment.901 }

      # Stellar AI: simple monthly surplus sales followed by shortage purchases.
      if = {
        limit = { is_ai = yes is_nomadic = no }
        set_variable = { which = stellarai_market_sale_count value = 0 }
        set_variable = { which = stellarai_market_purchase_count value = 0 }

        if = {
          limit = { is_at_war = no }
          if = {
            limit = { NOT = { stellarai_settled_ai_has_unemployment = yes } }
            inline_script = { script = stellarai/market_sell_surplus RESOURCE = minerals AMOUNT = 250 RESERVE = 2000 }
          }
          inline_script = { script = stellarai/market_sell_surplus RESOURCE = food AMOUNT = 250 RESERVE = 1500 }
          inline_script = { script = stellarai/market_sell_surplus RESOURCE = consumer_goods AMOUNT = 100 RESERVE = 1000 }
          inline_script = { script = stellarai/market_sell_surplus RESOURCE = volatile_motes AMOUNT = 10 RESERVE = 100 }
          inline_script = { script = stellarai/market_sell_surplus RESOURCE = exotic_gases AMOUNT = 10 RESERVE = 100 }
          inline_script = { script = stellarai/market_sell_surplus RESOURCE = rare_crystals AMOUNT = 10 RESERVE = 100 }
        }
        else = {
          if = {
            limit = { NOT = { stellarai_settled_ai_has_unemployment = yes } }
            inline_script = { script = stellarai/market_sell_surplus RESOURCE = minerals AMOUNT = 250 RESERVE = 1000 }
          }
          inline_script = { script = stellarai/market_sell_surplus RESOURCE = food AMOUNT = 250 RESERVE = 750 }
