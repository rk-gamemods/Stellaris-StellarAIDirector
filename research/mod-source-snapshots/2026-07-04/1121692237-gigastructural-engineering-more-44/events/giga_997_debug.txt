namespace = giga_debug

country_event = {
	id = giga_debug.1
	is_triggered_only = yes

	title = giga_debug.1.title
	desc = giga_debug.1.desc

	immediate = {
		set_variable = { which = total_galaxy_bm value = 0 }
		set_variable = { which = total_galaxy_systems value = 0 }
		every_system  = {
			calculate_system_bm = yes
			ROOT = {
				change_variable = {
					which = total_galaxy_systems
					value = 1
				}
				change_variable = {
					which = total_galaxy_bm
					value = PREV.giga_system_bm
				}
			}
		}
		set_variable = { which = average_galaxy_bm value = total_galaxy_bm }
		divide_variable = {
			which = average_galaxy_bm
			value = total_galaxy_systems
		}
	}
}
