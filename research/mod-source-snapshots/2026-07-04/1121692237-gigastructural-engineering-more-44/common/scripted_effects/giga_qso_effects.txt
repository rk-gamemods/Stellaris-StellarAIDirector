giga_qso_setup_owner = {
    if = {
        limit = {
            NOT = {
                has_country_flag = giga_qso_no_owner
            }
        }
        random_country = {
            limit = {
                has_country_flag = giga_qso_target_country@root
            }
            save_global_event_target_as = giga_qso_target_country
        }
    }
}

giga_qso_option_internal_if = {
    if = {
        limit = {
            NOT = {
                has_country_flag = giga_qso_no_owner
            }
        }
        event_target:giga_qso_target_country = {
            if = {
                limit = {
                    NOT = {
                        is_country_type = primitive
                    }
                }
                every_system_within_border = {
                    limit = {
                        qso_can_target_system = yes
                        or = {
                            [[condition]	$condition|always$ = yes ]
                            [[condition2]
                                $condition2$ = yes
                            ]
                            [[condition3]
                                $condition3$ = yes
                            ]
                            [[condition4] $condition4$ = yes  ]
                        }
                    }
                    giga_event_target_list_save = yes
                }
            }
            else = {
                random_owned_planet = {
                    solar_system = {
                        giga_event_target_list_save = yes
                    }
                }
            }
        }
    }
    else = {
        every_system = {
            limit = {
                qso_can_target_system = yes
                NOT = {
                    exists = owner
                }
                or = {
                    [[condition]	$condition|always$ = yes ]
                    [[condition2]
                        $condition2$ = yes
                    ]
                    [[condition3]
                        $condition3$ = yes
                    ]
                    [[condition4] $condition4$ = yes]
                }
            }
            giga_event_target_list_save = yes
        }
    }
}

giga_qso_target_cluster = {
    save_global_event_target_as = giga_qso_targeted_cluster
    set_planet_flag = giga_qso_targeted_cluster
    solar_system = {
        every_neighbor_system = {
            limit = {
                qso_can_target_system = yes
            }
            star = {
                set_planet_flag = giga_qso_targeted_cluster
            }
        }
    }
}

giga_qso_untarget_cluster = {
    event_target:giga_qso_targeted_cluster = {
        clear_global_event_target = giga_qso_targeted_cluster
        remove_planet_flag = giga_qso_targeted_cluster
        every_neighbor_system = {
            limit = {
                star = {
                    giga_qso_is_cluster_targeted = yes
                }
            }
            star = { remove_planet_flag = giga_qso_targeted_cluster }
        }
    }
}

# System
giga_qso_target_system = {
    from = {
        if = {
            limit = { NOT = { is_variable_set = giga_qso_targeted_systems } }
            set_variable = {
                which = giga_qso_targeted_systems
                value = 0
            }
        }
    }

    if = {
        limit = {
            from = {
                check_variable = { which = giga_qso_targeted_systems value = 0 }
            }
        }
        save_global_event_target_as = giga_qso_targeted_system_1
        set_planet_flag = giga_qso_targeted_system_1
        from = {
            change_variable = { which = giga_qso_targeted_systems value = 1 }
        }
    }
    else_if = {
        limit = {
            from = {
                check_variable = { which = giga_qso_targeted_systems value = 1 }
            }
        }
        save_global_event_target_as = giga_qso_targeted_system_2
        set_planet_flag = giga_qso_targeted_system_2
        from = {
            change_variable = { which = giga_qso_targeted_systems value = 1 }
        }
    }
    else_if = {
        limit = {
            from = {
                check_variable = { which = giga_qso_targeted_systems value = 2 }
            }
        }
        save_global_event_target_as = giga_qso_targeted_system_3
        set_planet_flag = giga_qso_targeted_system_3
        from = {
            change_variable = { which = giga_qso_targeted_systems value = 1 }
        }
    }
    else_if = {
        limit = {
            from = {
                check_variable = { which = giga_qso_targeted_systems value = 3 }
            }
        }
        save_global_event_target_as = giga_qso_targeted_system_4
        set_planet_flag = giga_qso_targeted_system_4

        }
    }
}
