    remove_megastructure_flag = giga_vat_$vat$_grow_voidspawn
    remove_megastructure_flag = giga_vat_$vat$_grow_swarm_queen
    remove_megastructure_flag = giga_vat_$vat$_grow_tiyanki

    remove_megastructure_flag = giga_vat_$vat$_grow_cutholoid
    remove_megastructure_flag = giga_vat_$vat$_grow_brood_mother
    remove_megastructure_flag = giga_vat_$vat$_grow_warrior

    clear_variable = the_vat_$vat$_pixel
    clear_variable = the_vat_$vat$_pixel_primary
    clear_variable = giga_vat_$vat$_grow_time

    clear_variable = giga_the_vat_$vat$_time

    clear_variable = giga_grow_$vat$_cost_primary
    clear_variable = giga_grow_$vat$_cost_secondary

    clear_variable = giga_vat_$vat$_grow_amount
}
