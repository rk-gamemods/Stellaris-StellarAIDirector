kaiser_opinion_range_elated = {
	check_variable = {
		which = kaiser_opinion
		value >= 80
	}
}

kaiser_opinion_range_very_friendly = {
	check_variable = {
		which = kaiser_opinion
		value < 80
	}
	check_variable = {
		which = kaiser_opinion
		value >= 60
	}
}

kaiser_opinion_range_friendly = {
	check_variable = {
		which = kaiser_opinion
		value < 60

kaiser_can_demand_system = {
	NOR = {
		is_capital_system = yes #Can't demand if its a capital
		has_star_flag = kaiser_demands_system #Not already demanded
	}
	owner = {
		giga_is_standard_country_type = yes #Only demand from regular empires (kaiser isn't suicidal)
		NOR = {
			is_at_war_with = event_target:flusion_primitives_country #Not at war
			is_in_federation_with = event_target:flusion_primitives_country #Not allied (kaiser isn't THAT cocky)
			has_country_flag = recently_gave_kaiser_land #didn't already give land to the Kaiser
		}
		has_country_flag = kaiser_demand_target #Same guy as before
	}
}

is_allowed_for_kaiser_spawn = {
	NOR = {
		has_star_flag = empire_home_system
		has_star_flag = empire_cluster

		#MEM
		has_star_flag = mem_severance_relic_system

		# is_fe_cluster = yes
		AND = {
			exists = space_owner
			space_owner = { is_fallen_empire = yes }
		}
		any_system = {
			distance = {
				source = this
				min_jumps  = 1
				max_jumps = 2
				use_bypasses = no
			}
			OR = {
				has_star_flag = empire_home_system
				has_star_flag = empire_cluster
				# is_fe_cluster = yes
				AND = {
					exists = space_owner
					space_owner = { is_fallen_empire = yes }
				}
			}
		}
	}
}

is_forbidden_for_kaiser_starbase = {
	#not already owned
	exists = starbase
	exists = owner
	any_system_planet = {
		is_colony = yes
		owner = {
			NOT = { is_country_type = primitive }
		}
	}

	#no hostiles
	any_fleet_in_system = {
		controller = {
			OR = {
				is_at_war_with = event_target:flusion_primitives_country
				is_hostile = event_target:flusion_primitives_country
			}
		}
	}

	# kaiser shouldn't anger the xenophobe fallen empire unless he's at war with them
	any_neighbor_system = {
		exists = owner
		owner = {
			is_fallen_empire = yes
			has_ai_personality = fallen_empire_xenophobe
			not = { is_at_war_with = root }
		}
	}

	# In case shrouded Faust is accessible, no-one should be in there
	has_star_flag = eawaf_faust_killzone

	# Don't interrupt the Sirens if they're guaranteed
	AND = {
		has_global_flag = giga_eawaf_sirens_guaranteed
		has_star_flag = eawaf_system
	}

	# Wild space compatibility
	is_star_class = sc_dn
}
