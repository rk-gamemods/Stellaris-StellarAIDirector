entity = asteroid_cw_02_entity
construction_entity = asteroid_cw_02_entity
portrait = "GFX_megastructure_asteroid_manufactory"
construction_blocks_and_blocked_by = none
show_in_outliner = yes

upgrade_from = {
    $upgrades_from$
}
upgrade_desc = hide

outliner_trigger = {
    exists = owner
    NOT = { has_megastructure_flag = giga_outliner_hidden_by_@owner }
}

build_time = 900
resources = {
    category = giga_kilostructures
    cost = {
        unity = @giga_kilo_unity_cost
    }
    inline_script = {
        script = megastructures/generic_parts/giga_mega_alloy_cost
        alloys = 750
    }
    upkeep = {
        energy = 25
        alloys = 10
    }
}

potential = {
    NOT = { has_global_flag = asteroid_manufactory_disabled }
}
possible = {
    from = { has_technology = giga_tech_asteroid_manufactory }
}

country_modifier = {
    $modifier$
}

ai_weight = {
    factor = 10
}

on_build_complete = {
    fromfrom = {
        set_megastructure_flag = giga_outliner_hidden_by_@owner
    }
    save_event_target_as = giga_system
    if = { limit = { exists = fromfrom.planet } fromfrom.planet = { save_event_target_as = giga_planet } }
    from = {
        country_event = {
            id = giga_asteroid_industry.003
            scopes = {
                from = this
                fromfrom = root.fromfrom
            }
        }				# Notification

        if = {
            limit = {
                NOR = {
                    has_global_flag = giga_achievements_disabled
                    has_country_flag = giga_achievement_54
                }
            }
            set_country_flag = giga_achievement_54
            set_timed_country_flag = { flag = giga_achievement_54_notification days = 30 }
            giga_achievement_sound = yes
        }
    }
}

dismantle_time = 360

dismantle_potential = {
    always = yes
}

dismantle_possible = {
    can_dismantle_megastructure = {
        TECH = giga_tech_asteroid_manufactory
    }
    custom_tooltip = {
        fail_text = requires_not_at_$building$_cap
        NOT = {
            owner = {
                OR = {
                    # One planet origin building check
                    AND = {
                        is_giga_one_planet_origin = yes
                        num_buildings = {
                            type = $building$_one_planet
                            value >= modifier:$building$_country_max
                        }
                    }

                    # Regular origin building check
                    num_buildings = {
                        type = $building$
                        value >= modifier:$building$_country_max
                    }
                }
            }
        }
    }
}

on_dismantle_start = {
    fromfrom.planet = { set_planet_flag = giga_dismantle_not_cancelled planet_event = { id = giga_dismantle.004 days = 361 } }
}

on_dismantle_cancel = {
    fromfrom.planet = { remove_planet_flag = giga_dismantle_not_cancelled }
}

on_dismantle_complete = {
    from = {
        add_resource = {
            alloys = 750
            mult = modifier:megastructure_dismantle_refund_mult
        }
        if = {
            limit = {
                check_variable = {
                    which = giga_current_asteroid_manufactory
                    value >= 1
                }
            }
            change_variable = {
                which = giga_current_asteroid_manufactory
                value = -1
            }
        }
    }
}
