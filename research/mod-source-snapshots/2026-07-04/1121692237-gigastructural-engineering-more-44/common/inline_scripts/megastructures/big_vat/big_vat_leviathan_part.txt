inline_script = {
    script = megastructures/big_vat/big_vat_normal_part
    vat = $vat$
    type = primary
}

giga_vat_$vat$_switch = {
    potential = {
        is_scope_type = megastructure
        exists = from
        from = { is_same_value = prev.owner }
        giga_is_vat_level_$vat$ = yes
        NOT = { has_megastructure_flag = giga_vat_$vat$_in_use }
    }
    allow = {}
    effect = {
        custom_tooltip = giga_vat_switch_$vat$
        switch = {
            trigger = has_megastructure_flag
            giga_vat_$vat$_fauna = {
                remove_megastructure_flag = giga_vat_$vat$_fauna
                set_megastructure_flag = giga_vat_$vat$_leviathans
            }
            giga_vat_$vat$_leviathans = {
                remove_megastructure_flag = giga_vat_$vat$_leviathans
                set_megastructure_flag = giga_vat_$vat$_scourge
            }
            giga_vat_$vat$_scourge = {
                remove_megastructure_flag = giga_vat_$vat$_scourge
                set_megastructure_flag = giga_vat_$vat$_fauna
            }
        }
    }
}
