planet_event = { #Start event and Sanity check
    id = giga_vat_events.$vat$4001
    is_triggered_only = yes
    hide_window = yes
    trigger = { vat_$vat$_check = yes }
    immediate = {
        solar_system = {
            random_system_megastructure = {
                limit = { vat_$vat$_chk = yes }
                set_variable = { which = the_vat_$vat$_time_passed value = 0 }
                set_variable = { which = the_vat_$vat$_percent value = 0 }
                set_variable = { which = giga_the_vat_$vat$_time value = giga_vat_$vat$_grow_time }
                divide_variable = { which = giga_the_vat_$vat$_time value = $base_speed_reduction$ }
                divide_variable = { which = giga_the_vat_$vat$_time value = value:giga_vat_timer_math }
                set_variable = { which = giga_vat_$vat$_time_left value = giga_the_vat_$vat$_time }
                round_variable = giga_vat_$vat$_time_left
                round_variable = giga_the_vat_$vat$_time
                #life is ~~pain~~ wonderful!
            }
        }
        if = { #Quick sanity check / pause exploit prevention
            limit = { NOT = { root = { has_planet_flag = giga_vat_pause_protection_$vat$ } } }
            root = { set_timed_planet_flag = {flag = giga_vat_pause_protection_$vat$ days = 1 } }
            planet_event = {
                id = giga_vat_events.$vat$4002
                days = 1
            }
        }
    }
}

planet_event = { #Recursive daily check until it reaches a value = build time
    id = giga_vat_events.$vat$4002
    is_triggered_only = yes
    hide_window = yes
    trigger = { vat_$vat$_check = yes }
    immediate = {
        solar_system = {
            random_system_megastructure = {
                limit = { vat_$vat$_chk = yes NOT = { root = { has_planet_flag = giga_vat_pause_protection_$vat$ } } }
                if = {
                    root = { set_timed_planet_flag = { flag = giga_vat_pause_protection_$vat$ days = 1 } }
                    limit = { check_variable = { which = the_vat_$vat$_time_passed value < giga_the_vat_$vat$_time } }

                    change_variable = { which = the_vat_$vat$_time_passed value = 1 }

                    set_variable = { which = the_vat_$vat$_pixel value = the_vat_$vat$_time_passed }
                    set_variable = { which = the_vat_$vat$_pixel_primary value = the_vat_$vat$_time_passed }

                    divide_variable	= { which = the_vat_$vat$_pixel value = giga_the_vat_$vat$_time }
                    divide_variable	= { which = the_vat_$vat$_pixel_primary value = giga_the_vat_$vat$_time }

                    set_variable = { which = the_vat_$vat$_percent value = the_vat_$vat$_pixel }
                    set_variable = { which = the_vat_$vat$_percent_primary value = the_vat_$vat$_pixel }

                    multiply_variable = { which = the_vat_$vat$_percent value = 100 }

                    multiply_variable = { which = the_vat_$vat$_pixel value = 229 }
                    multiply_variable = { which = the_vat_$vat$_pixel_primary value = 476 }

                    set_variable = { which = giga_vat_$vat$_time_left value = giga_the_vat_$vat$_time }

                    subtract_variable = { which = giga_vat_$vat$_time_left value = the_vat_$vat$_time_passed }

                    ceiling_variable = the_vat_$vat$_pixel
                    floor_variable = the_vat_$vat$_percent
                    root = {
                        planet_event = {
                            id = giga_vat_events.$vat$4002
                            days = 1
                        }
                    }
                } else_if = {
                    limit = { check_variable = { which = the_vat_$vat$_time_passed value >= giga_the_vat_$vat$_time } }
                    clear_variable = the_vat_$vat$_pixel
                    clear_variable = the_vat_$vat$_time_passed
                    set_variable = { which = the_vat_$vat$_pixel value = 0 }
                    root = {
                        planet_event = {
                            id = giga_vat_events.$vat$0000
                        }
                    }
                }
            }
        }
    }
}
