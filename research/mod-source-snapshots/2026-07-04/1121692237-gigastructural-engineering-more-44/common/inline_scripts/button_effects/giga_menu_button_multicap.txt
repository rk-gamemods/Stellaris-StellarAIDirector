# click to enable
$BUTTON_EFFECT_PREFIX$_disabled = {
    potential = {
        exists = from
        from = {
            has_country_flag = $GIGA_PAGE_FLAG$
        }
        has_global_flag = $MEGA_FLAG_NAME_PREFIX$_disabled
    }
    allow = { always = yes }
    effect = {
        custom_tooltip = "giga_menu_enabled"
        remove_global_flag = $MEGA_FLAG_NAME_PREFIX$_disabled
        # SETTING A DEFAULT HERE
        # OR ELSE TURNING IT ON AFTER SELECTING A PRESET WHERE ITS DISABLED DOESNT SELECT A SETTING
        # IE: turning on planetcraft after selecting vanilla+
        if = {
            # if there isnt an existing limit
            limit = {
                NOR = {
                    has_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_1
                    has_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_2
                    has_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_3
                    has_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_r
                    has_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_u
                }
            }
            # set the base
            set_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_1
        }
        $OTHER_ENABLE_EFFECTS$
    }
}
# visual to show its at this setting

    potential = {
        exists = from

        NOT = { has_global_flag = $MEGA_FLAG_NAME_PREFIX$_disabled }
        has_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_1 }
    allow = {
        custom_tooltip = {
            fail_text = "must_not_have_started_game"
            NOT = { has_global_flag = giga_game_started }
        }
    }
    effect = {
        custom_tooltip = "giga_menu_1"
    }
}

# click to disable
$BUTTON_EFFECT_PREFIX$_enabled = {
    potential = {
        exists = from
        from = { has_country_flag = $GIGA_PAGE_FLAG$ }
        NOT = { has_global_flag = $MEGA_FLAG_NAME_PREFIX$_disabled }
    }
    allow = { always = yes }
    effect = {
        custom_tooltip = "giga_menu_disabled"
        set_global_flag = $MEGA_FLAG_NAME_PREFIX$_disabled
        $OTHER_DISABLE_EFFECTS$
    }
}
# click to choose this setting
$BUTTON_EFFECT_PREFIX$_1_disabled = {
    potential = {
        exists = from
        from = { has_country_flag = $GIGA_PAGE_FLAG$ }
        OR = {
            has_global_flag = $MEGA_FLAG_NAME_PREFIX$_disabled
            NOT = { has_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_1 }
        }
    }
    allow = {
        custom_tooltip = {
            fail_text = "must_not_have_started_game"
            NOT = { has_global_flag = giga_game_started }
        }
        hidden_trigger = {
            NOT = { has_global_flag = $MEGA_FLAG_NAME_PREFIX$_disabled }
        }
    }
    effect = {
        remove_flags_$MEGA_FLAG_NAME_PREFIX$ = yes
        set_global_flag = $MEGA_FLAG_NAME_PREFIX$_capped_1
    }
}
