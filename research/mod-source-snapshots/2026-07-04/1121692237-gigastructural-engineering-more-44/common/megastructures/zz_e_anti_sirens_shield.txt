anti_sirens_shield = {
	entity = "giga_eawaf_sirens_panopticon_entity"
	construction_entity = "giga_eawaf_sirens_panopticon_entity"
	# construction_blocks_and_blocked_by = none # BLOCKING TEST
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = no
	build_type = inside_gravity_well
	construction_blocks_and_blocked_by = self_type
	prerequisites = { giga_tech_eawaf_disenchanter_4 }

	outliner_trigger = {
		exists = owner
		NOT = { has_megastructure_flag = giga_outliner_hidden_by_@owner }
	}
	
	build_time = 2000
	resources = {
		category = megastructures
		cost = {
			unity = @giga_small_mega_unity_cost
			rare_crystals = 1000
			exotic_gases = 1000
			volatile_motes = 1000
		}
		inline_script = {
			script = megastructures/generic_parts/giga_mega_alloy_cost
			alloys = 10000
		}
		upkeep = {
			energy = 100
			rare_crystals = 3
			exotic_gases = 3
			volatile_motes = 3
		}
	}

	potential = {
		is_ai = no
		has_technology = giga_tech_eawaf_disenchanter_4
	}

	possible = {
		custom_tooltip = { fail_text = "requires_inside_border"		is_inside_border = from }
	}

	placement_rules = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_anti_sirens_shield"
			NOT = {
				has_megastructure = anti_sirens_shield
			}
		}
	}

	ai_weight = {
		weight = 0
	}

	on_build_complete = {
		save_event_target_as = giga_system
		if = { limit = { exists = fromfrom.planet } fromfrom.planet = { save_event_target_as = giga_planet } }
		fromfrom.planet = {
			if = { limit = { exists = orbital_station } orbital_station = { dismantle = yes } }
			giga_set_has_mega_flag = yes
		}
		if = {
			limit = {
				NOT = { has_global_flag = giga_eawaf_shroudsonic_enthraller_destroyed }
				exists = event_target:giga_eawaf_country
			}
			event_target:giga_eawaf_country = {
				giga_eawaf_apply_strife_field_effects = yes
			}
		}
	}
}

anti_sirens_shield_dismantle = {
	entity = ""
	portrait = "GFX_megastructure_construction_background"
	upgrade_from = { anti_sirens_shield }

	outliner_trigger = {
		exists = owner
		NOT = { has_megastructure_flag = giga_outliner_hidden_by_@owner }
	}
	
	build_time = 180
	resources = {
		category = giga_megastructures
		cost = {
			energy = 1000
			alloys = -10000
		}
	}

	possible = { 
		from = { 
			hidden_trigger = {
				NOT = {
					AND = {
						NOT = { has_global_flag = giga_eawaf_sirens_defeated }
						is_ai = yes # AI please, stop taking these down
					}
				}
			}
		}
	}

	ai_weight = { weight = 999 } #no need for them after the crisis is over
	on_build_complete = {
		remove_megastructure = fromfrom
		fromfrom.planet = {
			giga_clear_has_mega_flag = yes
		}
		if = {
			limit = {
				NOT = { has_global_flag = giga_eawaf_shroudsonic_enthraller_destroyed }
				exists = event_target:giga_eawaf_country
			}
			set_update_modifiers_batch = begin
			event_target:giga_eawaf_country = {
				giga_eawaf_apply_strife_field_effects = yes
			}
			set_update_modifiers_batch = end
		}
	}
}
