giga_megaview_manage = {
	potential = {
		is_scope_type = megastructure
		exists = from # the viewing player
		from = { is_same_value = prev.owner } # hidden if we're looking at someone else's mega
	}
	allow = {
		# display text when active
		custom_tooltip = {
			text = giga_megaview_manage
			always = yes
		}

		# disable if already open
		hidden_trigger = {
			exists = owner
			owner = {
				nor = {
					has_active_event = { giga_mega_management.001 }
					is_ai = yes # you should not be able to click this for the AI as an observer!
				}
			}
		}

		# only allowed for specific megas
		custom_tooltip_fail = {
			text = giga_megaview_manage_no_options

			is_scope_type = megastructure
			
			giga_is_manageable_mega = yes

			#always = no # re-enable when we're ready for stuff by commenting this line out
		}
	}
	effect = {
		# display text when disabled
		custom_tooltip = giga_megaview_manage

		# open the management window
		hidden_effect = {
			if = { # A bit of a dirty short for this
				limit = { root = { is_megastructure_type = lunar_disco_ball_2 } }
				root.planet = { save_global_event_target_as = giga_mega_manage_planet }
				owner = {
					country_event = {
						id = giga_disco.1000
						# scopes = {

						# }
					}
				}
			}
			else = {
				owner = {
					country_event = {
						id = giga_mega_management.001
						scopes = {
							from = root
						}
					}
				}
			}
		}
	}
}

giga_megaview_highlight_toggle_on = {
	potential = {
		is_scope_type = megastructure
		exists = from # the viewing player
		from = { is_same_value = prev.owner } # hidden if we're looking at someone else's mega
		exists = planet
		planet = {
			not = { has_modifier = giga_highlight_object }
		}
	}
	allow = {
		# display text when active
		custom_tooltip = {
			text = giga_megaview_highlight
			always = yes
		}
	}
	effect = {
		# display text when disabled
		custom_tooltip = giga_megaview_highlight

		# add modifier
		hidden_effect = {
			planet = {
				add_modifier = {
					modifier = giga_highlight_object
				}
			}
		}
	}
}
