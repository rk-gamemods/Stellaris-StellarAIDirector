giga_extra_crucible_drone = {

            icon = giga_extra_healthcare
        }
    }

    possible_pre_triggers = {
        has_owner = yes
        is_enslaved = no
        is_being_purged = no
        is_being_assimilated = no
        is_sapient = yes
    }

    possible_precalc = can_fill_drone_job

    possible = { can_think = yes }

    planet_modifier = {
        giga_logistic_growth_mult_extra_add = 2
    }

    triggered_planet_modifier = {
        potential = {
            always = yes
        }
        planet_amenities_add = 500
        mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
    }

    resources = {
        category = planet_doctors
        upkeep = {
            alloys = 1
        }
        upkeep = {
            trigger = {
                exists = planet
                planet = {
                    any_targeting_situation = {
                        is_situation_type = giga_situation_extra_growth
                        has_situation_flag = has_species
                        check_variable = {
                            which = total_growth
                            value > 0
                        }
                        event_target:giga_extra_growth_species@this = {
                            is_lithoid = no
                            has_trait = trait_cybernetic
                        }
                    }
                }
            }
            food = 5
        }
        upkeep = {
            trigger = {
                exists = planet
                planet = {
                    any_targeting_situation = {
                        is_situation_type = giga_situation_extra_growth
                        has_situation_flag = has_species
                        check_variable = {
                            which = total_growth
                            value > 0
                        }
                        event_target:giga_extra_growth_species@this = {
                            is_lithoid = yes
                            has_trait = trait_cybernetic
                        }
                    }
                }
            }
            minerals = 5
        }
    }

    weight = {
        weight = @spawner_drone_job_weight
        modifier = {
            factor = 0.01 # crisis purge
            exists = planet
            exists = planet.controller
            planet.controller = {
                OR = {
                    is_country_type = swarm
                    is_country_type = ai_empire
                }
            }
        }
    }

    inline_script = jobs/automodding_priority_charisma
}
