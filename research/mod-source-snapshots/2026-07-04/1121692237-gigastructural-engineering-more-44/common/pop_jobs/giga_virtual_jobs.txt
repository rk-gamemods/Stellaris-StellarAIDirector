giga_virtual_hell_inmate = {
	category = criminal
    auto_generate_description = no

	swappable_data = {
		default = {
			building_icon = building_crude_huts
		}
	}

	is_capped_by_modifier = yes
	contributes_to_diplo_weight = no

	possible_pre_triggers = {
		has_owner = yes
		is_sapient = yes
	}

	possible = {
		# check that this is not a robot in servitude
		NOT = {
			has_citizenship_type = {
				country = owner
				type = citizenship_robot_servitude
			}
		}
		owner = {
			is_gestalt = no
		}
	}

	resources = {
		category = planet_jobs

		upkeep = {
			energy = 2
		}
        # experimenting on the damned costs more
        upkeep = {
            trigger = {
                exists = owner
                owner = {
                    is_twisted_experimenters_empire = yes
                }
            }
            energy = 3
        }
	}

	pop_group_modifier = {
		pop_housing_usage_add = -1
		pop_happiness = -2
	}

	planet_modifier = {
		planet_crime_add = 8
	}

	country_modifier = {
		planet_crime_add = -4
        pop_government_ethic_attraction = 0.025
        pop_cat_ruler_political_power = 0.05
	}

    # experiment on the damned
    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_twisted_experimenters_empire = yes
            }
        }
        planet_experiment_engineer_produces_mult = 0.05
        planet_test_subjects_produces_mult = 0.05
    }

	weight = {
		weight = @criminal_job_weight

		modifier = {
			factor = 0.1
			exists = owner
			owner = { is_xenophobe = yes }
			has_citizenship_rights = yes
		}

		modifier = {
			factor = 5.0
			exists = owner
			owner = { is_xenophobe = yes }
			has_citizenship_rights = no
		}
	}
}
