        potential = {
            exists = owner

    }
    # 2 supetensiles (pre-job eff) if the mega cap is job based
    triggered_planet_modifier = {
        potential = {
            has_global_flag = @giga_amb_flag
        }
        planet_miners_giga_sr_amb_megaconstruction_produces_add = 0.2
    }

    included_building_sets = {
        mining
        fallen_empire_mining
        mining_automation
    }

    planet_modifier = {
        zone_building_slots_add = 3
    }

    ai_priority = 5

    swap_type = district_giga_squareworld_livingmetal
    swap_type_weight = 5

    zone_sets = {
        ring_world
    }

    inline_script = {
        script = zones/shared_ring_world_zone_modifiers
    }
}
