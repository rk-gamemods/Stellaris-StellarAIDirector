# prerequisite building that gets upgraded automatically
building_giga_maginot_stronghold = {
	base_buildtime = 720
	base_cap_amount = 1
	planetary_ftl_inhibitor = yes
	position_priority = 200
	#add_to_first_building_slot = yes

	building_sets = {
		government
		fortress
	}

	category = army

	allow = {
		has_upgraded_capital = yes
	}

	potential = {
		# Maginot World should always have the better version
		NOT = {
			is_giga_maginot_world = yes
			has_carrier_flag = giga_maginot_world
		}
	}

	destroy_trigger = {
		always = no
	}

	planet_modifier = {
		planet_orbital_bombardment_damage = -0.25
	}

	army_modifier = {
		# army buffs
		army_starting_experience_add = 50
		army_experience_gain_mult = 0.10
		army_defense_damage_mult = 0.10
		army_defense_health_mult = 0.10
		army_defense_morale_mult = 0.10
		army_collateral_damage_mult = -0.05
	}

	inline_script = {
		script = jobs/maginot_bunker_officer_add
		AMOUNT = 100
	}

	resources = {
		category = planet_buildings_strongholds
		cost = {
			minerals = 1000
			alloys = 100
		}
		upkeep = {
			energy = 10
			alloys = 1
		}
	}

	prerequisites = {
		"giga_tech_maginot_world"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maginot_bunker_officer_gestalt_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_maginot_bunker_officer_effect_desc
	}

	inline_script = {
		script = buildings/on_all_wilderness_buildings_districts
	}
}
