terraform_link = {
	from = "pc_molten"
	to = "pc_ll_torrid"
	resources = {
		category = terraforming

		inline_script = {
			script = terraform/triggered_costs
			ENERGY_COST = 12500
			BIOMASS_COST = 6250
		}
	}
	inline_script = {
		script = terraform/wilderness_biomass
	}
	duration = 3600
	ai_weight = { weight = 5 }
	potential = {
		has_infernals = no
		from = { has_modifier = megastructure_terraforming_candidate }
	}
	effect = {
		from = {
			save_event_target_as = giga_planet
			if = {
				limit = { has_modifier = megastructure_terraforming_candidate }
				remove_modifier = megastructure_terraforming_candidate
			}
			add_modifier = { modifier = lifeless_torrid }
			add_modifier = { modifier = lifeless_torrid_tectonics days = 1800 }
			planet_event = { id = giga_mega.8601 days = 1 }
			set_planet_flag = used_to_be_molten
			solar_system = {
				random_system_megastructure = {
					limit = {
						OR = {
							is_megastructure_type = terraform_barren_1
							is_megastructure_type = terraform_toxic_1
							is_megastructure_type = terraform_molten_1
						}
						planet = { is_same_value = PREVPREVPREV }
					}
					remove_megastructure = THIS
				}
				save_event_target_as = giga_system
				space_owner = {
					country_event = { id = giga_dialog.4004 }				# Notification
					if = {
						limit = {
							NOR = {
								has_global_flag = giga_achievements_disabled
								has_country_flag = giga_achievement_57
							}
						}
						set_country_flag = giga_achievement_57
						set_timed_country_flag = { flag = giga_achievement_57_notification days = 30 }
						giga_achievement_sound = yes
					}
				}
			}
			remove_planet_flag = giga_terraforming_megastructure
			remove_carrier_flag = has_megastructure
		}
	}
}
